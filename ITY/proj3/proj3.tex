\section{Algoritmy} \label{sec:algoritmy}
  Pokud budeme chtít vysázet algoritmus, mùžeme použít prostøedí
  \texttt{algorithm}\footnote{Pro nápovìdu, jak zacházet s~prostøedím \texttt{algorithm}, mùžeme
  zkusit tuhle stránku:\\
  http://ftp.cstug.cz/pub/tex/CTAN/macros/latex/contrib/algorithms/algorithms.pdf.}
  nebo \texttt{algorithm2e}\footnote{Pro \texttt{algorithm2e} zase tuhle:
  http://ftp.cstug.cz/pub/tex/CTAN/macros/latex/contrib/algorithm2e/algorithm2e.pdf.}.
  Pøíklad použití prostøedí \texttt{algorithm2e} viz Algoritmus \ref{alg:fastslam}.\\

  \algsetup{indent=2em}
  \begin{algorithm}[ht]
    \label{alg:fastslam}
    \SetKwFor{for}{for}{do}{end~for}
    \SetKwInOut{input}{Input}
    \SetKwInOut{output}{Output}
    \SetKw{return}{return}
	\SetNlSty{}{}{:}
	\SetInd{1em}{1em}
	\SetNlSkip{-1.33em}
	
    \caption{\textsc{Fast}SLAM}
    \input{$\left( X_{t-1},u_t,z_t \right)$}
    \output{$X_t$}
	
    \BlankLine
	\Indp \Indp
	
    $\overline{X_t} = X_t = 0$\\
	\SetAlgoNoLine
	\for{$k = 1$ to $M$}{
	  $x^{[k]}_{t} =  \mathrm{\mathit{sample\_motion\_model}}    \left( u_t,x^{[k]}_{t - 1} \right)$\\
    $\omega^{[k]} = \mathrm{\mathit{measurement\_model}}       \left( z_t,x^{[k]}_{t},m_{t-1} \right)$\\
	  $m^{[k]}_{t} =  \mathrm{\mathit{updated\_occupancy\_grid}} \left( z_t,x^{[k]}_{t},m^{[k]}_{t-1} \right)$\\
	  $\overline{X_t} = \overline{X_t} + \left\langle x^{[m]}_{x}, \omega^{[m]}_{t} \right\rangle$\\
	  }{}
	\for{$k = 1$ to $M$}{
	  draw $i$ with probability $\approx \omega^{[i]}_{t}$\\
	  add $\left\langle x^{[k]}_{x}, m^{[k]}_{t} \right\rangle$ to $X_t$
	}{}
	\return{$X_t$}
  \end{algorithm}
